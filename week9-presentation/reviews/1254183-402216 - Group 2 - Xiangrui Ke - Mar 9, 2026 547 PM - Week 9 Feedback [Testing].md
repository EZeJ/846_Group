### Problem A_1: [Counterexample-First Prompting Wastes Tokens on Mini-Autograd Gradient Tests]

**Task Description:**

Write discriminative `pytest` unit tests for a minimal scalar autograd engien (e.g., `Tensor.backward` ) such that tests pass for the intended contract and fail for common buggy-but-plausible implementations. The goal is to maximize mutation coverage per token and avoid redundant “variant enumeration” that does not produce orthogonal discriminators.

**Starter Code:**

```python
# minimal_autograd_bug_example.py
# A tiny scalar autograd engine with a specific bug:
# BUG: gradient contributions are OVERWRITTEN instead of ACCUMULATED.

from __future__ import annotations
from dataclasses import dataclass
from typing import Callable, List, Optional, Tuple

@dataclass
class Context:
    parents: Tuple["Tensor", ...]
    backward_fn: Callable[[float], Tuple[Optional[float], ...]]  # dL/dout -> (dL/dp1, dL/dp2, ...)

class Tensor:
    def __init__(self, data: float, requires_grad: bool = False):
        self.data = float(data)
        self.requires_grad = bool(requires_grad)
        self.grad: Optional[float] = None
        self._ctx: Optional[Context] = None

    def __repr__(self) -> str:
        return f"Tensor(data={self.data}, grad={self.grad}, requires_grad={self.requires_grad})"

    def __add__(self, other: "Tensor") -> "Tensor":
        return add(self, other)

    def __mul__(self, other: "Tensor") -> "Tensor":
        return mul(self, other)

    def backward(self, grad: float = 1.0) -> None:
        if not self.requires_grad and self._ctx is None:
            return

        topo: List[Tensor] = []
        visited = set()

        def dfs(t: Tensor) -> None:
            if id(t) in visited:
                return
            visited.add(id(t))
            if t._ctx is not None:
                for p in t._ctx.parents:
                    dfs(p)
            topo.append(t)

        dfs(self)

        self.grad = grad

        for t in reversed(topo):
            if t._ctx is None:
                continue
            if t.grad is None:
                continue

            grads_to_parents = t._ctx.backward_fn(t.grad)
            for p, g in zip(t._ctx.parents, grads_to_parents):
                if g is None or not p.requires_grad:
                    continue

                # BUG: overwrite instead of accumulate
                p.grad = g

def add(a: Tensor, b: Tensor) -> Tensor:
    out = Tensor(a.data + b.data, requires_grad=(a.requires_grad or b.requires_grad))

    def backward_fn(dout: float) -> Tuple[Optional[float], Optional[float]]:
        return (dout, dout)

    out._ctx = Context(parents=(a, b), backward_fn=backward_fn)
    return out

def mul(a: Tensor, b: Tensor) -> Tensor:
    out = Tensor(a.data * b.data, requires_grad=(a.requires_grad or b.requires_grad))

    def backward_fn(dout: float) -> Tuple[Optional[float], Optional[float]]:
        return (dout * b.data, dout * a.data)

    out._ctx = Context(parents=(a, b), backward_fn=backward_fn)
    return out
```

    "multiplier": 2
}

    # expensive computation

        self.cache = {}

    def run(self, x):
        if x in self.cache:
            return self.cache[x]
        result = compute(x)
        self.cache[x] = result
        return result

import re

def normalize_username(name: str) -> str:
    if name is None:
        raise TypeError("username cannot be None")

    name = name.strip().lower()

    if len(name) < 3 or len(name) > 30:
        raise ValueError("invalid length")

    if not re.match(r"^[a-z0-9_]+$", name):
        raise ValueError("invalid characters")

    return name
